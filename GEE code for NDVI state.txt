function applyScaleFactors(image) {
  var opticalBands = image.select('SR_B.').multiply(0.0000275).add(-0.2);
  var thermalBands = image.select('ST_B.*').multiply(0.00341802).add(149.0);
  return image.addBands(opticalBands, null, true)
              .addBands(thermalBands, null, true);
}


var processCountry = function(index) {
  var geometry = ee.Feature(list.get(index));
  var roi_geometry = geometry.geometry();
  
  var L8 = ee.ImageCollection('LANDSAT/LC08/C02/T1_L2')
                  .filterBounds(roi_geometry)
                  .filter(ee.Filter.calendarRange(2014, 2022, 'year'))
                  .map(applyScaleFactors)
                  .select(['SR_B4', 'SR_B5', 'QA_PIXEL'], ['red', 'nir', 'QA_PIXEL']);
                  
  var L7 = ee.ImageCollection('LANDSAT/LE07/C02/T1_L2')
                  .filterBounds(roi_geometry)
                  .filter(ee.Filter.calendarRange(2012, 2013, 'year'))
                  .map(applyScaleFactors)
                  .select(['SR_B3', 'SR_B4', 'QA_PIXEL'], ['red', 'nir', 'QA_PIXEL']);
  
  var L5 = ee.ImageCollection('LANDSAT/LT05/C02/T1_L2')
                  .filterBounds(roi_geometry)
                  .filter(ee.Filter.calendarRange(2007, 2011, 'year'))
                  .map(applyScaleFactors)
                  .select(['SR_B3', 'SR_B4', 'QA_PIXEL'], ['red', 'nir', 'QA_PIXEL']);
  
  var Landsat = ee.ImageCollection(L8.merge(L7).merge(L5)).sort("system:time_start");

  function rmCloudNew(image) {
    var cloudShadowBitMask = (1 << 4); 
    var cloudsBitMask = (1 << 3); 
    var qa = image.select('QA_PIXEL'); 
    var mask = qa.bitwiseAnd(cloudShadowBitMask).eq(0)
                  .and(qa.bitwiseAnd(cloudsBitMask).eq(0)); 
    return image.updateMask(mask)
                .copyProperties(image)
                .copyProperties(image, ["system:time_start"]);
  }
  var filtered = Landsat.map(rmCloudNew)

  var filtered = filtered.map(function(image) {
    var timeImage = image.metadata('system:time_start').rename('timestamp')
    // The time image doesn't have a mask. 
    // We set the mask of the time band to be the same as the first band of the image
    var timeImageMasked = timeImage.updateMask(image.mask().select(0))
    return image.addBands(timeImageMasked)
  })
  var days = 160
  var millis = ee.Number(days).multiply(1000*60*60*24)
  var maxDiffFilter = ee.Filter.maxDifference({
    difference: millis,
    leftField: 'system:time_start',
    rightField: 'system:time_start'
  })
  
  var lessEqFilter = ee.Filter.lessThanOrEquals({
    leftField: 'system:time_start',
    rightField: 'system:time_start'
  })
  
  var greaterEqFilter = ee.Filter.greaterThanOrEquals({
    leftField: 'system:time_start',
    rightField: 'system:time_start'
  })
  
  var filter1 = ee.Filter.and(maxDiffFilter, lessEqFilter)
  var join1 = ee.Join.saveAll({
    matchesKey: 'after',
    ordering: 'system:time_start',
    ascending: false})
    
  var join1Result = join1.apply({
    primary: filtered,
    secondary: filtered,
    condition: filter1
  })
  var filter2 = ee.Filter.and(maxDiffFilter, greaterEqFilter)
  var join2 = ee.Join.saveAll({
    matchesKey: 'before',
    ordering: 'system:time_start',
    ascending: true})
    
  var join2Result = join2.apply({
    primary: join1Result,
    secondary: join1Result,
    condition: filter2
  })
  
  var interpolateImages = function(image) {
    var image = ee.Image(image)
    var beforeImages = ee.List(image.get('before'))
    var beforeMosaic = ee.ImageCollection.fromImages(beforeImages).mosaic()
    var afterImages = ee.List(image.get('after'))
    var afterMosaic = ee.ImageCollection.fromImages(afterImages).mosaic()
  
    var t1 = beforeMosaic.select('timestamp').rename('t1')
    var t2 = afterMosaic.select('timestamp').rename('t2')
  
    var t = image.metadata('system:time_start').rename('t')
  
    var timeImage = ee.Image.cat([t1, t2, t])
  
    var timeRatio = timeImage.expression('(t - t1) / (t2 - t1)', {
      't': timeImage.select('t'),
      't1': timeImage.select('t1'),
      't2': timeImage.select('t2'),
    })
    
    var interpolated = beforeMosaic
      .add((afterMosaic.subtract(beforeMosaic).multiply(timeRatio)))
    // Replace the masked pixels in the current image with the average value
    var result = image.unmask(interpolated)
    return result.copyProperties(image, ['system:time_start'])
  }
  
  // map() the function to interpolate all images in the collection
  var interpolatedCol = ee.ImageCollection(join2Result.map(interpolateImages))
  
  function addNDVI(image) {
    var ndvi = image.normalizedDifference(['nir','red']).rename('ndvi');
    return image.addBands(ndvi);
  }
  var visCollectionInterpolated = interpolatedCol
    .map(addNDVI)
    
  //***2007-2019***//
  //   //juheweinianduzhi  
  var startYear = 2007;
  var endYear = 2019;
  var years = ee.List.sequence(startYear, endYear);
  var collectyear = ee.ImageCollection(years  
    .map(function(year) {
    var start = ee.Date.fromYMD(year, 4, 1);
    var end = ee.Date.fromYMD(year, 10, 30);
    // Filter the image collections for the year
    var landsat = visCollectionInterpolated.filterDate(start, end).select('ndvi');
    // var modis = MOD13Q1_ndvi.filterDate(start, end).mean().reproject('EPSG:4326', null, 250);
    // Blend, reduce, and clip the images
    return landsat.reduce(ee.Reducer.mean()).float().rename('mean_ndvi').clip(roi_geometry) 
                .set('year', year)
                .set('system:time_start', start.millis());
  }));
  
   
  var processImage = function(image) {
    var year = ee.Number.parse(image.get('system:index')); 
    var yearImage = ee.Image.constant(year).toShort();
    return ee.Image.cat(yearImage, image).rename(['year', 'data']).set('year', year);
  };
  
  var processedCol = collectyear.map(processImage);
  // print('Annual datasets', processedCol);
  

  var meanImage = processedCol.select('data').mean();
  var stdDevImage = processedCol.select('data').reduce(ee.Reducer.stdDev());
  
  //***2019-2022***//
  var startYear1 = 2020;
  var endYear1 = 2022;
  var years1 = ee.List.sequence(startYear1, endYear1);
  var collectyear1 = ee.ImageCollection(years1  
    .map(function(year) {
    var start1 = ee.Date.fromYMD(year, 4, 1);
    var end1 = ee.Date.fromYMD(year, 10, 30);
    // Filter the image collections for the year
    var landsat1 = visCollectionInterpolated.filterDate(start1, end1).select('ndvi');
    // var modis = MOD13Q1_ndvi.filterDate(start, end).mean().reproject('EPSG:4326', null, 250);
    // Blend, reduce, and clip the images
    return landsat1.reduce(ee.Reducer.mean()).float().rename('mean_ndvi1').clip(roi_geometry) 
                .set('year', year)
                .set('system:time_start', start1.millis());
  }));
   
  var processImage1 = function(image) {
    var year = ee.Number.parse(image.get('system:index'));
    var yearImage = ee.Image.constant(year).toShort();
    return ee.Image.cat(yearImage, image).rename(['year', 'data1']).set('year', year);
  };
  
  var processedCol1 = collectyear1.map(processImage1);
  var meanImage1 = processedCol1.select('data1').mean();

  // ***calculate Z***//
  var sqrt3 = ee.Number(3).sqrt();
  var zImage = meanImage1.subtract(meanImage).divide(stdDevImage.multiply(sqrt3));
  
  var classifiedZImage = zImage.where(zImage.gte(1.96), 1) 
                              .where(zImage.lte(-1.96), -1) 
                              .where(zImage.gt(-1.96).and(zImage.lt(1.96)), 2) 
                              .int();
  var sig1 = classifiedZImage.clip(roi_geometry)
  
  Export.image.toDrive({
	image:sig1,
	description:'ndvi_states_2007_2022_' + i,
	folder:'NDVI_states',
	region:roi_geometry,
	scale:30,
	crs:"EPSG:4326",
	maxPixels:1e13,
	fileFormat:'GeoTIFF',
  })
};

var n = roi.size().getInfo();
for (var i = 0; i < n; i++) {  
  processCountry(i);
}
